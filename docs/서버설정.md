# 운영서버 환경변수 설정시
> ### 서버 터미널에서 아래와 같이 환경 변수를 설정합니다.
> ### 보통 ~/.bashrc 나 /etc/profile 파일에 추가하여 영구적으로 설정합니다.
> export DB_USER='prod_user'
> export DB_PASSWORD='very_strong_prod_password'
> export MYSQL_ROOT_PASSWORD='very_strong_root_password'
> export JWT_SECRET_KEY='your_super_secret_key_for_production_!@#$'
> 
> 
> 로그확인
> docker logs cs-chat-root-cs-app-1
> 엔진엑스 sudo tail -f /var/log/nginx/error.log
> 
> 
> 
> 
## 1. GitHub에서 최신 버전의 docker-compose를 다운로드하여 /usr/local/bin 에 저장
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

## 2. 다운로드한 파일에 실행 권한 부여
sudo chmod +x /usr/local/bin/docker-compose

# 1. 서버에 업로드한 프로젝트 폴더로 이동합니다.
cd /home/your_user/cs-chat

# 2. 이 위치에서 다시 명령어를 실행합니다.
docker-compose up -d
